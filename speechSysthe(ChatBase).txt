import { CommandResult } from '../types';

let model: any = null;
try {
  
  const apiKey = import.meta.env.VITE_CHATBASE_API_KEY;  
  if (apiKey) {
   
    chatbaseAI = new ChatbaseAI(apiKey);  
    model = chatbaseAI.getGenerativeModel();  
  }
} catch (error) {
  console.warn('Chatbase AI initialization failed:', error);
}
